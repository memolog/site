function insertionSort(array, showArray) {
  let count = 0;
  const arrayLength = array.length;
  for (let i=1; i<arrayLength; i++) {
    const a = array[i];
    let j = i-1;
    for (; j>=0; j--) {
      count++;
      const b = array[j];
      if (a < b) {
        array[j+1] = b
      } else {
        break;
      }
    }
    array[j+1] = a;
    if (showArray) {
      console.log(array);
    }
  }
  console.log(`number of iterates: ${count}`);
}
